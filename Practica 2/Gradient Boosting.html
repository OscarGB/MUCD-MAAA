La primera, createDataSet, es para crear los problemas, siempre con dos clases y en dos dimensiones. Sus argumentos son:

def createDataSet(n,model,ymargin,noise=None,output_boundary=False):
    x = np.random.rand(n,1)*2.0*np.pi
    xbnd = np.linspace(0,2.0*np.pi,100)

    if model == 'sine':
        y = (np.random.rand(n,1) - 0.5)*2.2
        c = y > np.sin(x)
        ybnd = np.sin(xbnd)
    elif model == 'linear':
        y = np.random.rand(n,1)*2.0*np.pi
        c = y > x
        ybnd = xbnd
    elif model == 'square':
        y = np.random.rand(n,1)*4.0*np.pi*np.pi
        c = y > x*x
        ybnd = xbnd*xbnd
    else:
        y = np.random.rand(n,1)*2.0*np.pi
        c = y > x
        ybnd = xbnd
    
    y[c == True] = y[c == True] + ymargin
    y[c == False] = y[c == False] - ymargin
    
    if noise is not None:
        y = y + noise * np.random.randn(n,1)
        x = x + noise * np.random.randn(n,1)

    if output_boundary == True:
        return x, y, (c*1).ravel(), xbnd, ybnd
    else:
        return x, y, (c*1).ravel()

def plotModel(x,y,clase,clf,title=""):
    from matplotlib.colors import ListedColormap
    
    x_min, x_max = x.min() - .2, x.max() + .2
    y_min, y_max = y.min() - .2, y.max() + .2
    hx = (x_max - x_min)/100.
    hy = (y_max - y_min)/100.
    xx, yy = np.meshgrid(np.arange(x_min, x_max, hx), np.arange(y_min, y_max, hy))

    if hasattr(clf, "decision_function"):
        z = clf.decision_function(np.c_[xx.ravel(), yy.ravel()])
    elif hasattr(clf, "predict_proba"):
        z = clf.predict_proba(np.c_[xx.ravel(), yy.ravel()])[:, 1]
    else:
        z = clf.predict(np.c_[xx.ravel(), yy.ravel()])

    z = z.reshape(xx.shape)
    cm = plt.cm.RdBu
    cm_bright = ListedColormap(['#FF0000', '#0000FF'])
    
    plt.contourf(xx, yy, z, cmap=cm, alpha=.8)
    plt.contour(xx, yy, z, [0.5], linewidths=[2], colors=['k'])

    if clase is not None:
        plt.scatter(x[clase==0], y[clase==0], c='#FF0000')
        plt.scatter(x[clase==1], y[clase==1], c='#0000FF')
    else:
        plt.plot(x,y,'g', linewidth=3)
        
    plt.gca().set_xlim(xx.min(), xx.max())
    plt.gca().set_ylim(yy.min(), yy.max())
    plt.grid(True)
    plt.xlabel("X")
    plt.ylabel("Y")
    plt.title(title)

from sklearn.tree import DecisionTreeClassifier
import numpy as np

class BaggingCasero:
    def __init__(self, n_estimators=101):
        self.n_estimators = n_estimators
        self._estimators = []
    
    def fit(self,X,y):
        N = X.shape[0]
        for i in range(self.n_estimators):
            """
              Rellenar esta parte para que cree un rbol usando
              una muestra bootstrap de los datos
            """
            # Bootstrap
            bootstrap_indices = np.random.choice(list(range(N)), size=N, replace=True)
            X_b = X[bootstrap_indices]
            y_b = y[bootstrap_indices]
            # Entrenar un rbol con la muestra bootstrap
            tree = DecisionTreeClassifier()
            tree.fit(X_b, y_b)
            self._estimators.append(tree)
            
    def predict(self,X):
        votos = np.zeros((X.shape[0],len(self._estimators)))
        # Calcula la salida de cada rbol para cada dato
        for ie,estimator in enumerate(self._estimators):
            votos[:,ie] = estimator.predict(X)
        
        """
           Calcula la clase ms votada de cada ejemplo, es decir,
           la moda
        """
        moda = stats.mode(votos, axis=1)[0]
        return moda

class SquaredErrorLoss:
    """ Clase que define elementos para la funcin cuadrtica de perdida 
        para regresin """
    
    def F0(_, X, y):
        """ Calcula el valor constate que minimiza la salida 'y' """
        return np.mean(y)

    def residuos(_, y, F):
        """ Calcula los residuos para un objetivo 'y' y 
            una salida del modelo F """
        return y - F

    ## CAMBIO 01
    def paso_newton_hoja(_, y, residuos, valor):
        """ Funcin para actualizar la salida de una hoja el rbol
        
           Recibe informacin sobre los ejemplos de una hoja dada. En concreto
           para los ejemplos que caen en una hoja:
              * El vector de valores a predecir (y)
              * Los pseudo-residuos (residuos) sobre los que se ha
                entrenado alarbol regresor ht
           Adems recibe el valor actual de salida de la hoja
              
            Debe devolver el valor actualizado
        """
        return 1
    
    def __call__(self, y, F):
        """ Devuelve el valor de la funcin de prdida para un objetivo 'y' y 
            una salida del modelo F """
        return 0.5*(y-F)**2
    
    def func(self, F):
        """ Devuelve el valor de la prediccin dependiendo 
            del score dado por los rboles
        """
        return F
    
class LogLoss:
    """ Clase que define elementos para la funcin logstica de perdida 
        para clasificacin de dos clases {-1, +1} """
    
    def F0(_, X, y):
        """ Calcula el valor constate que minimiza la salida 'y' """
        ybar = np.mean(y)
        return 0.5*np.log((1+ybar)/(1-ybar))

    def residuos(_, y, F):
        """ Calcula los residuos para un objetivo 'y' y 
            una salida del modelo F """
        return 2*y/(1 + np.exp(2*y*F))

    ## CAMBIO 02
    def paso_newton_hoja(_, y, residuos, valor):
        """ Funcin para actualizar la salida de una hoja el rbol
        
           Recibe informacin sobre los ejemplos de una hoja dada. En concreto
           para los ejemplos que caen en una hoja:
              * El vector de valores a predecir (y)
              * Los pseudo-residuos (residuos) sobre los que se ha
                entrenado alarbol regresor ht
           Adems recibe el valor actual de salida de la hoja
              
            Debe devolver el valor actualizado
        """
        abr = np.abs(residuos)
        rho =  np.sum(residuos)/np.sum(abr*(2-abr))
        return rho
    
    def __call__(self,y, F):
        """ Devuelve el valor de la funcin de prdida para un objetivo 'y' y 
            una salida del modelo F """
        return np.log(1+np.exp(-2.0*y*F))
    
    def func(self, F):
        """ Devuelve el valor de la prediccin (1 o -1) dependiendo 
            del score dado por los rboles
        """
        probs = 1 / (1 + np.exp(-2*F))
        probs[probs < 0.5] = -1
        probs[probs >= 0.5] = 1
        return probs
     
class GBCasero:
    def __init__(self, n_estimators=101, loss=SquaredErrorLoss(), eta=0.1, depth=3):
        self.n_estimators = n_estimators
        self._estimators  = []
        self.depth        = depth
        self.eta          = eta # Learning rate
        # La implementacin de loss se usar en fit para crear el algoritmo
        # GB de forma genrica.
        self.loss         = loss
    
    def fit(self,X,y):
        """
          Inicializa GB 
        """
        self.F0 = self.loss.F0(X, y)
        # Prediccin inicial
        F = self.F0
        for i in range(self.n_estimators):
            tree = DecisionTreeRegressor(max_depth = self.depth)
            """
              Rellenar esta parte para implementar GB 
              Se debe utilizar la funcin paso_newton de abajo
            """
            r = self.loss.residuos(y, F)
            # Entrenar el arbol con los residuos obtenidos del modelo del paso anterior
            tree.fit(X, r)
            self.paso_newton_general(tree, X, y, r)
            # Actualizar la prediccin
            F = F + self.eta*tree.predict(X)
            # Aadirlo a la lista de modelos
            self._estimators.append(tree)
            
    def predict(self,X):
        """ Combinamos los valores de pesos y hs para obtener la 'salida', y:
               * Para regresin se devuelve 'salida'
               * Para clasificacin de 2 clases se calcula la probabilidad
                  con una sigmoidal (1 / (1 + np.exp(-2*salida))) y se 
                  devuelve la clase ms probable """
        F = self.F0 + self.eta*np.sum([tree.predict(X) for tree in self._estimators], axis=0)
        # Dependiendo de la funcin de prdida hay que realizar ajuste
        return self.loss.func(F)
            
    ## CAMBIO 03
    def paso_newton_general(self, tree, X, y, residuos):
        """ Esta funcin actualiza todas las hojas de salida
            del rbol 'tree' utilizado la funcin paso_newton_hoja
            de las clases de funcion de prdida """
        TREE_LEAF = -1
        tree = tree.tree_
        leaf_indices = tree.apply(np.array(X,dtype=np.float32))
        for leaf in np.where(tree.children_left == TREE_LEAF)[0]:
            ii = leaf_indices==leaf
            tree.value[leaf,0,0] = self.loss.paso_newton_hoja(y[ii],
                                                              residuos[ii],
                                                              tree.value[leaf,0,0])

# individualPredictions: Devuelve la prediccin para cada dato por parte de cada clasificador 
#                        de un conjunto de clasificadores
#     Entrada:
#         - ens: lista con un conjunto de clasificadores
#         - X  : ejemplos a clasificar
#     Salida:
#         - Matriz de predicciones de nmero de ejemplo filas y no. clasificadores columnas
def individualPredictions(ens, X):
    P = np.ones((X.shape[0],len(ens)))
    it = 0
    for dt in ens:
        P[:,it] = dt.predict(X)
        it += 1

    return P

def accuracy(ens, pr, y):
    pr = np.cumsum(pr,axis=1)/np.arange(1.,pr.shape[1]+1,1.)
    pr[pr>0.5] = 1
    pr[pr<=0.5] = 0

    iclases_test[y==ens.classes_[0]] = 0
    iclases_test[y==ens.classes_[1]] = 1

    Pok = pr==iclases_test
    return np.array(Pok.sum(axis=0),dtype=float)/len(y)
